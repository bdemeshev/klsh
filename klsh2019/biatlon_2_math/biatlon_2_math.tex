\begin{enumerate}

\item Запиши данное число максимально просто
\[
\sqrt{8 + 2\sqrt{8 + 2\sqrt{8 + 2\sqrt{\ldots}}}}
\]

\item Запиши данное число максимально просто
\[
\frac{8}{2 + \frac{8}{2 + \frac{8}{2 + \frac{8}{2 + \ldots}}}}
\]

\end{enumerate}

\begin{enumerate}
    \item Решаем:
    \[
    \sqrt{8 + 2x} = x    
    \]
    После возведения в квадрат получаем два корня, $x=4$, $x=-2$. Оставляем корень $4$.

    \item Решаем 
    \[
    \frac{8}{2 + x} = x    
    \]
    После возведения в квадрат получаем два корня: $2$, $-4$. Оставляем корень $2$. 
\end{enumerate}
